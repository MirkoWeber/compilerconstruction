// Grammatik Start

S -> PC ID PB | PC PB
PC -> keyword"Package" identifier
ID -> keyword"import" string
PB -> FN
FN -> keyword"func" identifier '(' ')' '{' FB '}'
FB -> identifier '=' AE | keyword"if" LE ST | identifier FC | ';' FB | null 

FC -> '.' identifier FC |  '(' AG 
AG -> identifier AG' | const AG' | ')'
AG' -> ',' AG 

AE -> TE AE'		//Taschenrechner Beispiel
AE' -> '+' TE AE' | null
TE -> FA TE'
TE' -> '*' FA TE' | null
FA -> '(' AE ')' | identifier | constant

// Grammatik Ende

// nur für lexxer

string = "{letter | number}" 
identifier = letter { letter | number }
tokens = ( | ) | ; | { | }
